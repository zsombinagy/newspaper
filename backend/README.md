# Setup

1. npm init -y
2. npm i -D typescript
3. ni script.ts
4. npx tsc --init
5. tsconfig.json l58 - "outDir": "./out",
6. .gitignore - node_modules out
7. package.json - "build": "tsc"
8. npm run build
9. npm run start

# After clone

1. npm i
2. npm run build
3. npm run start
4. rm -Force .git
5. git init...
